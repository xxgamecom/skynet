# source file list
AUX_SOURCE_DIRECTORY("${SKYNET_ROOT_PATH}/skynet-service-c/snlua" SNLUA_SRC)

)

    ARCHIVE_OUTPUT_DIRECTORY "${SKYNET_CSERVICE_BIN_PATH}"

)
TARGET_LINK_LIBRARIES(snlua liblua)
